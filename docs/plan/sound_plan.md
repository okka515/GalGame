## 卒業できるの、できないの！？ 6人組最後の一年 サウンド演出プラン

このゲームの「ドタバタな青春群像劇・コメディ」という軸と、「友情・進路・恋愛・研究」というテーマを重視し、プレイヤーが感情移入しつつも、時にコミカルに、時に真剣に物語を楽しめるようなサウンドデザインを目指します。特に、コメディ要素を際立たせるためのSEや、シーンの切り替えを明確にするBGMの活用を意識しました。

### NarraLeaf-Reactの `Sound` クラスを使ったコードへの組み込み例

NarraLeaf-Reactでは、`Sound` クラスを使ってBGMやSEを管理・再生できます。ここでは、いくつかのパターンでの組み込み例を提示します。

```typescript
// src/game.ts または該当のシナリオファイル (例: src/scenarios/opening.ts)

import { Scenario, Sound } from 'narraleaf'; // NarraleafのSoundクラスをインポート

// BGMインスタンスの作成（ゲーム起動時や初期化時に一度行う）
// 卒業パーティのような賑やかで、しかし少し感傷的なBGM
const graduationPartyBGM = new Sound({ src: '/public/audio/bgm_graduation_party.mp3', loop: true, volume: 0.6 });
// コミカルな通知音
const notificationSFX = new Sound({ src: '/public/audio/sfx_notification_comical.mp3' });
// 好感度アップのキラキラSE
const affectionUpSFX = new Sound({ src: '/public/audio/sfx_affection_up.mp3' });
// コミカルなツッコミSE
const tsukkomiSFX = new Sound({ src: '/public/audio/sfx_tsukkomi.mp3' });


// シナリオの例
export const openingScenario: Scenario = {
  id: 'opening',
  pages: [
    {
      text: '（カレンダーをめくる音）',
      onEnter: () => { /* 他の演出（背景変更など） */ },
      effects: [
        { type: 'sound', action: 'play', sound: new Sound({ src: '/public/audio/sfx_calendar_flip.mp3' }) },
      ],
    },
    {
      text: '「いよいよ最終学年の4月。気合入れていくぞ！」',
      speaker: '主人公',
      onEnter: () => {
        // プロローグBGMの再生
        graduationPartyBGM.play();
      },
    },
    {
      text: 'その矢先、スマホが次々と通知を告げた。',
      speaker: '主人公',
      onEnter: () => {
        // BGMを維持しつつ、通知音を連続再生
        notificationSFX.play();
        setTimeout(() => notificationSFX.play(), 500);
        setTimeout(() => notificationSFX.play(), 1000);
      },
    },
    {
      text: 'ぱっく「おい、聞いてくれ！フランス人の彼女が…」',
      speaker: 'ぱっく',
      onEnter: () => {
        // BGMを切り替え
        graduationPartyBGM.stop(); // 現在のBGMを停止
        const comicalBGM = new Sound({ src: '/public/audio/bgm_comical_trouble.mp3', loop: true, volume: 0.7 });
        comicalBGM.play();
      }
    },
    {
      text: '「お前、またとんでもないことになってるのか！」',
      speaker: '主人公',
      effects: [
        { type: 'sound', action: 'play', sound: tsukkomiSFX }, // ツッコミSEの再生
      ],
    },
    {
      text: '（ここでプレイヤーの選択肢：ぱっくの相談に乗るか？）',
      choices: [
        {
          text: 'ぱっくの相談に乗る',
          nextScene: 'pack_route_start',
          onChoose: () => {
            // 好感度上昇演出
            affectionUpSFX.play();
            // BGMを再度切り替える場合
            // comicalBGM.stop();
            // graduationPartyBGM.play();
          }
        },
        { text: '他の誰かの相談に乗る', nextScene: 'choose_other_character' },
      ],
    },
    // ...以降のシナリオ
  ],
};
```
